$(document).ready(function () { var i, timerId, speed = 100, //speed in ms higlightWord = function (text, index) { var words = text.match(/\w+/g); if (index >= words.length) { return text; } words.splice(index, 0, ""); words.splice(index + 2, 0, ""); return words.join(' '); }; $('.highlightButton').bind('click', function () { $inputText = $('#inputText'); i = 0; if (timerId) { clearInterval(timerId); } timerId = setInterval(function () { var sentence = higlightWord($inputText.text(), i); if (sentence !== $inputText.text()) { i += 1; } else { clearInterval(timerId); } $inputText.html(sentence); }, speed); }); });
